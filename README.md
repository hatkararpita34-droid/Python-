# # Declare variables with different data types

# Integer variable
# Integers are whole numbers
integer_var = 10

# Float variable
# Floats are numbers with decimal points
float_var = 20.5

# String variable
# Strings are sequences of characters, enclosed in quotes
string_var = "Hello, Python!"

# Boolean variable
# Booleans represent True or False values
boolean_var = True

# List variable
# Lists are ordered collections of items
list_var = [1, 2, 3, "four", 5.0]

# Tuple variable
# Tuples are ordered, immutable collections of items
tuple_var = (10, 20, 30, "forty")

# Dictionary variable
# Dictionaries are collections of key-value pairs
dict_var = {"name": "Alice", "age": 30}


# Display variables and their types

print(f"Variable: {integer_var}, Type: {type(integer_var)}")
print(f"Variable: {float_var}, Type: {type(float_var)}")
print(f"Variable: {string_var}, Type: {type(string_var)}")
print(f"Variable: {boolean_var}, Type: {type(boolean_var)}")
print(f"Variable: {list_var}, Type: {type(list_var)}")
print(f"Variable: {tuple_var}, Type: {type(tuple_var)}")
print(f"Variable: {dict_var}, Type: {type(dict_var)}")
